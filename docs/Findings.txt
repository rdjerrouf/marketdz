create or replace function public.build_multilingual_tsquery(
  p_search_text text,
  p_lang text default 'any',
  p_max_tokens int default 6,
  p_max_expansions int default 6
)
returns table(q_ar tsquery, q_fr tsquery)
language plpgsql
stable
as $$
declare
  v_token text;
  v_terms text[];
  v_group_ar tsquery;
  v_group_fr tsquery;
  v_all_ar tsquery;
  v_all_fr tsquery;
  v_tokens text[];
begin
  v_tokens := regexp_split_to_array(lower(coalesce(p_search_text, '')), '\\s+');

  v_tokens := (
    select coalesce(array_agg(t), '{}'::text[])
    from (
      select t
      from unnest(v_tokens) as t
      where length(t) >= 2
      limit greatest(1, p_max_tokens)
    ) s
  );

  if coalesce(array_length(v_tokens, 1), 0) = 0 then
    return query select null::tsquery, null::tsquery;
    return;
  end if;

  foreach v_token in array v_tokens loop
    select coalesce(
      array(
        select distinct x
        from (
          select v_token as x
          union all
          select unnest(l.equivalents)
          from public.search_lexicon l
          where l.is_active = true
            and l.term = v_token
            and (l.lang = p_lang or l.lang = 'any')
        ) z
        where length(x) >= 2
        limit greatest(1, p_max_expansions)
      ),
      array[v_token]
    )
    into v_terms;

    v_group_ar := null;
    v_group_fr := null;

    for i in 1..coalesce(array_length(v_terms, 1), 0) loop
      v_group_ar := case when v_group_ar is null
        then to_tsquery('arabic', quote_literal(v_terms[i]))
        else v_group_ar || to_tsquery('arabic', quote_literal(v_terms[i])) end;

      v_group_fr := case when v_group_fr is null
        then to_tsquery('french', quote_literal(v_terms[i]))
        else v_group_fr || to_tsquery('french', quote_literal(v_terms[i])) end;
    end loop;

    v_all_ar := case when v_all_ar is null then v_group_ar else v_all_ar && v_group_ar end;
    v_all_fr := case when v_all_fr is null then v_group_fr else v_all_fr && v_group_fr end;
  end loop;

  return query select v_all_ar, v_all_fr;
end;
$$;

    )::real as rank_score,
    'fts'::text as match_source
  from public.listings l
  cross join q
  where l.status = 'active'
    and (
      (q.q_ar is not null and l.search_vector_ar @@ q.q_ar)
      or
      (q.q_fr is not null and l.search_vector_fr @@ q.q_fr)
    )
),
fallback as (
  select
    l.id,
    l.title,
    l.description,
    l.company_name,
    l.price,
    l.status,
    l.created_at,
    extensions.word_similarity(lower(coalesce(l.title, '')), lower(coalesce(p_search_text, '')))::real as rank_score,
    'trgm_fallback'::text as match_source
  from public.listings l
  where l.status = 'active'
    and not exists (select 1 from fts_hits)
    and lower(coalesce(l.title, '')) % lower(coalesce(p_search_text, ''))
    and extensions.word_similarity(lower(coalesce(l.title, '')), lower(coalesce(p_search_text, ''))) >= p_min_trgm_similarity
)
select *
from (
  select * from fts_hits
  union all
  select * from fallback
) s
order by s.rank_score desc, s.created_at desc
limit greatest(1, least(p_limit, 100))
offset greatest(0, p_offset);
$$;
